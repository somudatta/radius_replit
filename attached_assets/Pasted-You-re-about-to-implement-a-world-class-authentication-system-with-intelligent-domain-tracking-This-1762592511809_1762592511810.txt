```javascript
// lib/auth.js
import { createClient } from '@supabase/supabase-js'

const supabase = createClient(
  process.env.NEXT_PUBLIC_SUPABASE_URL,
  process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY
)

export const authService = {
  // Sign Up
  async signUp(email, password, name) {
    try {
      const { data, error } = await supabase.auth.signUp({
        email,
        password,
        options: {
          data: {
            name,
            created_at: new Date().toISOString(),
            plan: 'free', // Default plan
            domains_analyzed: 0
          }
        }
      })
      
      if (error) throw error
      
      // Create user profile in database
      if (data.user) {
        await this.createUserProfile(data.user.id, { email, name })
      }
      
      return { success: true, user: data.user }
    } catch (error) {
      return { success: false, error: error.message }
    }
  },

  // Sign In
  async signIn(email, password) {
    try {
      const { data, error } = await supabase.auth.signInWithPassword({
        email,
        password
      })
      
      if (error) throw error
      
      // Log sign in event
      await this.logUserActivity(data.user.id, 'sign_in')
      
      return { success: true, user: data.user }
    } catch (error) {
      return { success: false, error: error.message }
    }
  },

  // Sign Out
  async signOut() {
    const { error } = await supabase.auth.signOut()
    if (!error) {
      window.location.href = '/'
    }
  },

  // Get Current User
  async getCurrentUser() {
    const { data: { user } } = await supabase.auth.getUser()
    return user
  },

  // Create User Profile
  async createUserProfile(userId, userData) {
    const { error } = await supabase
      .from('users')
      .insert({
        id: userId,
        ...userData,
        created_at: new Date().toISOString()
      })
    
    if (error) console.error('Profile creation error:', error)
  }
}
```

### 5. DOMAIN TRACKING SERVICE
```javascript
// lib/domainHistory.js
export const domainHistoryService = {
  // Save domain to history when analyzed
  async saveDomain(userId, domain, analysisData) {
    try {
      const { data, error } = await supabase
        .from('domain_history')
        .insert({
          user_id: userId,
          domain: domain,
          analysis_data: analysisData,
          ai_visibility_score: analysisData.overallScore,
          improvements_found: analysisData.improvements.length,
          status: 'completed'
        })
        .select()
        .single()

      if (error) throw error

      // Save individual platform results
      if (data && analysisData.platforms) {
        await this.savePlatformResults(data.id, analysisData.platforms)
      }

      return { success: true, data }
    } catch (error) {
      console.error('Error saving domain:', error)
      return { success: false, error }
    }
  },

  // Get user's domain history
  async getUserDomains(userId, limit = 20, offset = 0) {
    try {
      const { data, error, count } = await supabase
        .from('domain_history')
        .select('*', { count: 'exact' })
        .eq('user_id', userId)
        .order('analyzed_at', { ascending: false })
        .range(offset, offset + limit - 1)

      if (error) throw error

      return { 
        success: true, 
        domains: data, 
        total: count,
        hasMore: count > offset + limit
      }
    } catch (error) {
      return { success: false, error }
    }
  },

  // Quick re-analyze
  async reanalyzeDomain(userId, domain) {
    // Check if domain was analyzed recently (within 24 hours)
    const { data: recent } = await supabase
      .from('domain_history')
      .select('*')
      .eq('user_id', userId)
      .eq('domain', domain)
      .gte('analyzed_at', new Date(Date.now() - 24 * 60 * 60 * 1000).toISOString())
      .single()

    if (recent) {
      return {
        success: true,
        data: recent,
        cached: true,
        message: 'Using recent analysis (less than 24 hours old)'
      }
    }

    // Perform new analysis
    return this.analyzeDomain(domain, userId)
  },

  // Search domains
  async searchDomains(userId, searchTerm) {
    const { data, error } = await supabase
      .from('domain_history')
      .select('*')
      .eq('user_id', userId)
      .ilike('domain', `%${searchTerm}%`)
      .order('analyzed_at', { ascending: false })
      .limit(10)

    return { success: !error, data, error }
  }
}
```

```javascript
// middleware.js (Next.js)
import { createMiddlewareClient } from '@supabase/auth-helpers-nextjs'
import { NextResponse } from 'next/server'

export async function middleware(req) {
  const res = NextResponse.next()
  const supabase = createMiddlewareClient({ req, res })
  
  const {
    data: { session },
  } = await supabase.auth.getSession()

  // Protect dashboard and analysis routes
  const protectedRoutes = ['/dashboard', '/analysis', '/settings']
  const isProtectedRoute = protectedRoutes.some(route => 
    req.nextUrl.pathname.startsWith(route)
  )

  if (isProtectedRoute && !session) {
    // Redirect to login with return URL
    const redirectUrl = new URL('/login', req.url)
    redirectUrl.searchParams.set('returnUrl', req.nextUrl.pathname)
    return NextResponse.redirect(redirectUrl)
  }

  return res
}

export const config = {
  matcher: ['/dashboard/:path*', '/analysis/:path*', '/settings/:path*']
}
```
